1 · Overview & Architecture

This tutorial shows you how to go from the BlueBush repository to a production-quality AI chatbot (and later a full-duplex voice bot) for the BlueBush sustainable homewares store. The approach uses a two-layer AI strategy:

  • SFT / LoRA fine-tuning — teaches the LLM the BlueBush persona, tone, and conversational style ("Aussie-casual, eco-savvy").
  • RAG (Retrieval-Augmented Generation) — injects live business knowledge (products, FAQs, orders) at inference time so product data never goes stale.

Keep these two layers distinct: fine-tuning is never the right tool for injecting factual product data (it memorises poorly and is expensive to update). RAG is the right tool for facts; fine-tuning is the right tool for how the bot speaks.

2 Convert BlueBush-Training-Dataset.JSON to Chat-ML JSONL

The JSON training dataset uses a flat {"customer_inquiry","brand_response"} schema. We must convert it to the same chat-ML format used by the primary dataset before merging. 

import json, pathlib

SYSTEM_PROMPT = (
    "You are the BlueBush Virtual Assistant — a knowledgeable, eco-conscious, "
    "and friendly shopping guide for an Australian sustainable homewares store. "
    "Respond in a warm 'Aussie-casual' tone: friendly and professional. "
    "Use natural Australian phrasing (e.g., 'No worries', 'Cheers', 'Sorted') "
    "but avoid tourist clichés. Keep answers concise and helpful."
)

raw_path  = pathlib.Path("source/BlueBush-Training-Dataset.JSON")
out_path  = pathlib.Path("/tmp/bluebush_extra.jsonl")

with raw_path.open() as f:
    raw_examples = json.load(f)   # list of 53 dicts

converted = []
for ex in raw_examples:
    inquiry  = ex.get("customer_inquiry", "").strip()
    response = ex.get("brand_response",   "").strip()
    if not inquiry or not response:
        continue
    converted.append({
        "messages": [
            {"role": "system",    "content": SYSTEM_PROMPT},
            {"role": "user",      "content": inquiry},
            {"role": "assistant", "content": response},
        ]
    })

with out_path.open("w") as f:
    for row in converted:
        f.write(json.dumps(row) + "\n")

print(f"Converted {len(converted)} examples → {out_path}")

# ── Simple evaluation harness ─────────────────────────────────────
# After fine-tuning, run this to get a rough BLEU / ROUGE baseline.
# For production, use human evaluation with Australian-English speakers.

def evaluate_model(model, tokenizer, eval_ds, n=50):
    """Quick automated eval — ROUGE-L on assistant turns."""
    from rouge_score import rouge_scorer
    scorer  = rouge_scorer.RougeScorer(["rougeL"], use_stemmer=True)
    scores  = []
    sample  = eval_ds.shuffle(seed=0).select(range(min(n, len(eval_ds))))
    for row in sample:
        messages = row["messages"]
        prompt_msgs  = [m for m in messages if m["role"] != "assistant"]
        reference    = next(m["content"] for m in messages if m["role"] == "assistant")
        input_text   = tokenizer.apply_chat_template(
            prompt_msgs, tokenize=False, add_generation_prompt=True
        )
        inputs   = tokenizer(input_text, return_tensors="pt").to(model.device)
        outputs  = model.generate(**inputs, max_new_tokens=200, do_sample=False)
        predicted = tokenizer.decode(outputs[0][inputs["input_ids"].shape[1]:],
                                     skip_special_tokens=True)
        scores.append(scorer.score(reference, predicted)["rougeL"].fmeasure)
    avg = sum(scores) / len(scores)
    print(f"ROUGE-L avg over {len(scores)} examples: {avg:.3f}")
    return avg

4 · Fine-Tuning with LoRA (SFT — Persona & Style)

What SFT teaches: How BlueBush speaks — Aussie warmth, eco vocabulary, short voice-optimised sentences, conflict-resolution scripts. It does not teach product facts (that's RAG).

3 Tokenise Dataset

MAX_LEN = 1024   # reduce to 512 if OOM on T4

def tokenise(example):
    text = tokenizer.apply_chat_template(
        example["messages"], tokenize=False, add_generation_prompt=False
    )
    encoded = tokenizer(
        text,
        truncation=True,
        max_length=MAX_LEN,
        padding="max_length",
    )
    # Labels: -100 for prompt tokens so loss is computed on assistant only
    labels = encoded["input_ids"].copy()
    # Find where assistant turn starts and mask everything before it
    for i, msg in enumerate(example["messages"]):
        if msg["role"] == "assistant":
            break
    prompt_only = tokenizer.apply_chat_template(
        example["messages"][:i],
        tokenize=False, add_generation_prompt=True
    )
    prompt_len = len(tokenizer(prompt_only, truncation=True,
                               max_length=MAX_LEN)["input_ids"])
    labels[:prompt_len] = [-100] * prompt_len
    encoded["labels"] = labels
    return encoded

tok_train = train_ds.map(tokenise, remove_columns=train_ds.column_names,
                         num_proc=2)
tok_val   = val_ds.map(tokenise,   remove_columns=val_ds.column_names,
                       num_proc=2)

5 · RAG Indexing (Business Knowledge)

What RAG handles: Live product catalogue, stock levels, prices, FAQs, order status — anything that changes frequently and must never be fabricated. RAG is served at inference time; no retraining needed when products change.

1 Enable pgvector in Supabase

-- Run in Supabase SQL Editor (Dashboard → SQL Editor → New Query)
CREATE EXTENSION IF NOT EXISTS vector;

-- ── Embeddings table ─────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS embeddings (
    id         BIGSERIAL PRIMARY KEY,
    doc_id     TEXT        NOT NULL,         -- e.g. "product:BED-001"
    source     TEXT        NOT NULL,         -- "products", "faqs", "orders"
    content    TEXT        NOT NULL,         -- full text chunk
    embedding  vector(384) NOT NULL,         -- 384-dim for all-MiniLM-L6-v2
    metadata   JSONB,
    created_at TIMESTAMPTZ DEFAULT now()
);

-- ── HNSW index for fast approximate nearest-neighbour search ──────
CREATE INDEX IF NOT EXISTS idx_embeddings_hnsw
    ON embeddings USING hnsw (embedding vector_cosine_ops)
    WITH (m = 16, ef_construction = 64);

-- ── RLS: allow service-role reads (gateway), deny anon ───────────
ALTER TABLE embeddings ENABLE ROW LEVEL SECURITY;
-- (No anon policy = anon cannot read embeddings; service-role always bypasses RLS)

2 Export Supabase Tables & Build Documents

from supabase import create_client, Client
import json, pathlib, os

SUPABASE_URL     = os.environ["SUPABASE_URL"]
SUPABASE_SVC_KEY = os.environ["SUPABASE_SERVICE_KEY"]  # service-role (server only)

sb: Client = create_client(SUPABASE_URL, SUPABASE_SVC_KEY)

def fetch_all(table: str, columns: str = "*") -> list[dict]:
    """Fetch all rows from a Supabase table (handles pagination)."""
    rows, start = [], 0
    while True:
        resp = sb.table(table).select(columns).range(start, start + 999).execute()
        rows.extend(resp.data)
        if len(resp.data) < 1000:
            break
        start += 1000
    return rows

products = fetch_all("products")
faqs     = fetch_all("faqs")
print(f"Fetched: {len(products)} products, {len(faqs)} FAQs")

# ── Also load the local product catalogue as fallback ──────────────
catalog_path = pathlib.Path("source/BlueBush-Product-Catalog-V1.json")
if catalog_path.exists():
    with catalog_path.open() as f:
        local_catalog = json.load(f)
    print(f"Local catalogue: {len(local_catalog)} products")

def product_to_text(p: dict) -> str:
    """Convert a product row into a retrievable text chunk."""
    lines = [
        f"Product: {p.get('name', p.get('id', ''))}",
        f"Category: {p.get('category', '')}",
        f"Price: ${p.get('price_aud', p.get('price', 0)):.2f} AUD",
        f"Stock: {p.get('stock_level', p.get('stock', 'unknown'))} units",
        f"Description: {p.get('description', p.get('marketing_hook', ''))}",
        f"Materials: {p.get('origin', p.get('materials', ''))}",
        f"Eco badge: {p.get('eco_badge', '')}",
        f"Care: {p.get('care_instructions', '')}",
        f"Variants: {', '.join(p.get('variants', []))}",
    ]
    return "\n".join(l for l in lines if l.split(': ', 1)[-1].strip())

def faq_to_text(f: dict) -> str:
    return f"Question: {f.get('question', '')}\nAnswer: {f.get('answer', '')}"

# ── Build document list ────────────────────────────────────────────
documents = []
for p in (products or local_catalog):
    documents.append({
        "doc_id":  f"product:{p.get('id', p.get('name', ''))}",
        "source":  "products",
        "content": product_to_text(p),
        "metadata": {"id": p.get("id"), "category": p.get("category")},
    })
for fq in faqs:
    documents.append({
        "doc_id":  f"faq:{fq.get('id', fq.get('question', '')[:40])}",

3 Build Embeddings & Upsert to pgvector

from sentence_transformers import SentenceTransformer
import numpy as np

# all-MiniLM-L6-v2: 384-dim, fast, good quality; free, no API key needed
EMBED_MODEL = "sentence-transformers/all-MiniLM-L6-v2"
embedder    = SentenceTransformer(EMBED_MODEL)

BATCH_SIZE = 64
rows_to_insert = []

contents = [d["content"] for d in documents]
embeddings = embedder.encode(
    contents, batch_size=BATCH_SIZE, show_progress_bar=True, normalize_embeddings=True
)

for doc, emb in zip(documents, embeddings):
    rows_to_insert.append({
        "doc_id":    doc["doc_id"],
        "source":    doc["source"],
        "content":   doc["content"],
        "embedding": emb.tolist(),   # pgvector accepts a list of floats
        "metadata":  doc["metadata"],
    })

# ── Upsert in batches (delete-then-insert for idempotency) ────────
for i in range(0, len(rows_to_insert), 100):
    batch = rows_to_insert[i:i+100]
    # Delete existing docs with same doc_ids
    doc_ids = [r["doc_id"] for r in batch]
    sb.table("embeddings").delete().in_("doc_id", doc_ids).execute()
    sb.table("embeddings").insert(batch).execute()

print(f"Indexed {len(rows_to_insert)} documents into Supabase pgvector ✓")

4 Retrieval Function (used by AI Gateway)

def retrieve_context(query: str, top_k: int = 3) -> str:
    """
    Retrieve top_k most relevant chunks for a user query.
    Call this from the AI Gateway before assembling the final prompt.
    """
    query_emb = embedder.encode([query], normalize_embeddings=True)[0].tolist()
    # Use Supabase's built-in RPC for vector similarity search
    resp = sb.rpc(
        "match_embeddings",           # see SQL function below
        {"query_embedding": query_emb, "match_count": top_k}
    ).execute()
    chunks = [row["content"] for row in resp.data]
    return "\n\n---\n".join(chunks)

-- Add this RPC function to Supabase (SQL Editor)
CREATE OR REPLACE FUNCTION match_embeddings(
    query_embedding vector(384),
    match_count     INT DEFAULT 3
)
RETURNS TABLE (
    id       BIGINT,
    doc_id   TEXT,
    source   TEXT,
    content  TEXT,
    similarity FLOAT
)
LANGUAGE SQL STABLE
AS $$
    SELECT id, doc_id, source, content,
           1 - (embedding <=> query_embedding) AS similarity
    FROM embeddings
    ORDER BY embedding <=> query_embedding
    LIMIT match_count;
$$;

Retrieval + Grounding Approach in the Gateway

The gateway assembles the final prompt as follows:

┌─ System prompt (from source/BlueBush_Persona_Config.md) ───────────────┐
│ Role: You are the BlueBush Virtual Assistant …                          │
│                                                                         │
│ [CONTEXT DATA START]                                                    │
│ {retrieved_chunks — top-3 nearest from pgvector}                       │
│ [CONTEXT DATA END]                                                      │
└─────────────────────────────────────────────────────────────────────────┘
  + conversation history (last 5 turns)
  + current user message

Keep retrieved context under ~800 tokens to stay within the context window of a Llama 3 8B model running at typical gateway timeout limits. Use only the top-3 chunks; more rarely helps and adds latency.

Cloudflare Worker — Gateway Skeleton

// gateway/src/index.js  (Cloudflare Worker)
// Deploy: npx wrangler deploy
//
// Environment variables (set via `wrangler secret put` or CF dashboard):
//   LLM_API_KEY        — HF Inference API or OpenAI-compatible endpoint key
//   LLM_ENDPOINT       — e.g. https://api-inference.huggingface.co/models/your-username/bluebush-lora
//   SUPABASE_URL       — https://<project-id>.supabase.co
//   SUPABASE_SVC_KEY   — service-role key (NEVER expose to browser)
//   GATEWAY_API_KEY    — a secret string the website JS must include in X-API-Key header
//   EMBED_ENDPOINT     — optional: hosted embedding service URL

const ALLOWED_ORIGINS = ["https://azzamunza.github.io"];  // GitHub Pages origin

export default {
  async fetch(request, env) {
    // ── CORS preflight ────────────────────────────────────────────
    if (request.method === "OPTIONS") {
      return corsHeaders(new Response(null, { status: 204 }), request, env);
    }

    // ── Auth: require X-API-Key header ────────────────────────────
    const apiKey = request.headers.get("X-API-Key");
    if (apiKey !== env.GATEWAY_API_KEY) {
      return corsHeaders(new Response("Unauthorised", { status: 401 }), request, env);
    }

    const url = new URL(request.url);

    if (url.pathname === "/chat" && request.method === "POST") {
      return corsHeaders(await handleChat(request, env), request, env);
    }

    return corsHeaders(new Response("Not found", { status: 404 }), request, env);
  }
};

async function handleChat(request, env) {
  const { message, history = [] } = await request.json();

  // ── 1. RAG: retrieve context from Supabase pgvector ─────────────
  const context = await retrieveContext(message, env);

  // ── 2. Assemble prompt ────────────────────────────────────────────
  const systemPrompt = `You are the BlueBush Virtual Assistant — eco-conscious,
friendly, Aussie-casual. Use the context below to answer product questions.
Never invent prices or stock levels.

[CONTEXT DATA START]
${context}
[CONTEXT DATA END]`;

  const messages = [
    { role: "system",    content: systemPrompt },
    ...history.slice(-10),   // last 5 turns (10 messages)
    { role: "user",      content: message },
  ];

  // ── 3. Call fine-tuned LLM (OpenAI-compatible endpoint) ──────────
  const llmResp = await fetch(env.LLM_ENDPOINT, {
    method:  "POST",
    headers: {
      "Content-Type":  "application/json",
      "Authorization": `Bearer ${env.LLM_API_KEY}`,
    },
    body: JSON.stringify({ messages, max_tokens: 400, temperature: 0.7 }),
  });
  const llmData  = await llmResp.json();
  const reply    = llmData.choices?.[0]?.message?.content ?? "No worries — let me look into that for you.";

  return new Response(JSON.stringify({ reply }), {
    headers: { "Content-Type": "application/json" },
  });
}

async function retrieveContext(query, env) {
  // Call your embedding service to get query embedding, then Supabase RPC
  // Simplified version using Supabase REST + a serverless embed endpoint:
  try {
    const embedResp = await fetch(env.EMBED_ENDPOINT, {
      method:  "POST",
      headers: { "Content-Type": "application/json" },
      body:    JSON.stringify({ text: query }),
    });
    const { embedding } = await embedResp.json();

    const sbResp = await fetch(
      `${env.SUPABASE_URL}/rest/v1/rpc/match_embeddings`,
      {
        method:  "POST",
        headers: {
          "Content-Type":  "application/json",
          "apikey":        env.SUPABASE_SVC_KEY,
          "Authorization": `Bearer ${env.SUPABASE_SVC_KEY}`,
        },
        body: JSON.stringify({ query_embedding: embedding, match_count: 3 }),
      }
    );
    const chunks = await sbResp.json();
    return chunks.map(c => c.content).join("\n\n---\n");
  } catch (err) {
    console.warn("[RAG] Retrieval failed, answering without context:", err);
    return "";   // graceful degradation — answer without RAG context
  }
}

function corsHeaders(response, request, env) {
  const origin = request.headers.get("Origin") || "";
  if (ALLOWED_ORIGINS.includes(origin)) {
    response.headers.set("Access-Control-Allow-Origin", origin);
    response.headers.set("Access-Control-Allow-Methods", "POST, OPTIONS");
    response.headers.set("Access-Control-Allow-Headers", "Content-Type, X-API-Key");
  }
  return response;
}

async function checkRateLimit(ip, env) {
  const key     = `rl:${ip}`;
  const current = parseInt(await env.RATE_LIMIT_KV.get(key) || "0");

}

8 · Voice — Australian Neutral + PersonaPlex Full-Duplex

Important distinction. The accent and voice identity is the job of the TTS or voice model. The vocabulary and phrasing ("No worries", "Cheers", "Sorted") is the job of the text LLM (trained in Section 4). These are separate concerns — do not conflate them.

Phase A — MVP: Text Chat + Non-Full-Duplex TTS/ASR

Deliver a working voice experience without any specialised voice infrastructure. Use the browser Web Speech API (ASR) and a TTS service. 

// js/voice-mvp.js — Browser Web Speech API (ASR) + TTS
// No new backend needed; works with the existing AI Gateway /chat endpoint

// ── ASR: listen ─────────────────────────────────────────────────
const recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
recognition.lang     = "en-AU";   // Australian English
recognition.interimResults = false;

recognition.onresult = async (event) => {
  const userText = event.results[0][0].transcript;
  const reply    = await sendMessage(userText);  // existing chat function
  speakReply(reply);
};

// ── TTS: speak ───────────────────────────────────────────────────
function speakReply(text) {
  const utterance      = new SpeechSynthesisUtterance(text);
  utterance.lang       = "en-AU";
  // Try to select an Australian voice if available
  const voices         = window.speechSynthesis.getVoices();
  const auVoice        = voices.find(v => v.lang === "en-AU") || null;
  if (auVoice) utterance.voice = auVoice;
  utterance.rate  = 1.0;
  utterance.pitch = 1.0;
  window.speechSynthesis.speak(utterance);
}

Server-side TTS alternative (Phase A+): Route TTS through the gateway to use OpenAI TTS (tts-1 with alloy or nova voice) or ElevenLabs. This gives much higher quality without requiring PersonaPlex.

Phase B — Full Duplex: NVIDIA PersonaPlex

PersonaPlex is a single unified model that handles ASR → LLM → TTS in one low-latency stream (<300 ms). It uses a dual-stream architecture: audio-in and text-in are processed concurrently, and audio-out is streamed back as tokens are generated.

Gateway WebSocket Handler (PersonaPlex streaming)

# gateway/voice_session.py  (Python, e.g., FastAPI on Cloud Run)
# This is where PersonaPlex makes Cloud Run a better fit than CF Workers
# (WebSocket + streaming binary audio is complex in Workers)

import asyncio, os, websockets
from fastapi import FastAPI, WebSocket

NVIDIA_API_KEY      = os.environ["NVIDIA_API_KEY"]
PERSONAPLEX_WS_URL  = "wss://integrate.api.nvidia.com/v1/personaplex/stream"
SYSTEM_PROMPT       = open("system_prompt.txt").read()

app = FastAPI()

@app.websocket("/voice")
async def voice_session(ws: WebSocket, api_key: str):
    """Full-duplex voice bridge: Browser ↔ Gateway ↔ PersonaPlex."""
    if api_key != os.environ["GATEWAY_API_KEY"]:
        await ws.close(code=4001)
        return
    await ws.accept()

    async with websockets.connect(
        PERSONAPLEX_WS_URL,
        extra_headers={"Authorization": f"Bearer {NVIDIA_API_KEY}"}
    ) as pp_ws:
        # ── Send session config ──────────────────────────────────
        import json
        await pp_ws.send(json.dumps({
            "type":          "session.init",
            "system_prompt": SYSTEM_PROMPT,
            "voice_preset":  "en-AU-neutral",   # AU neutral voice preset
            "lora_adapter":  "your-username/bluebush-lora",   # HF Hub adapter ID
            "sample_rate":   24000,
        }))

        # ── Bidirectional relay ────────────────────────────────────
        async def browser_to_pp():
            async for audio_chunk in ws.iter_bytes():
                await pp_ws.send(audio_chunk)

        async def pp_to_browser():
            async for message in pp_ws:
                if isinstance(message, bytes):
                    await ws.send_bytes(message)   # audio tokens → browser
                else:
                    data = json.loads(message)
                    if data.get("type") == "transcript":
                        await ws.send_json(data)   # live transcript → UI

        await asyncio.gather(browser_to_pp(), pp_to_browser())

// js/voice-fullduplex.js
const ws = new WebSocket(
  `wss://bluebush-gateway.run.app/voice?api_key=${window.GATEWAY_API_KEY}`
);
ws.binaryType = "arraybuffer";

// ── Microphone capture (PCM 24 kHz, mono) ──────────────────────
const stream  = await navigator.mediaDevices.getUserMedia({ audio: true });
const ctx     = new AudioContext({ sampleRate: 24000 });
const source  = ctx.createMediaStreamSource(stream);
const proc    = ctx.createScriptProcessor(2048, 1, 1);
// NOTE: ScriptProcessorNode is deprecated; for production, replace with AudioWorkletNode
// (see https://developer.mozilla.org/en-US/docs/Web/API/AudioWorkletNode)

proc.onaudioprocess = (e) => {
  const pcm = e.inputBuffer.getChannelData(0);
  ws.send(Float32Array.from(pcm).buffer);
};
source.connect(proc);
proc.connect(ctx.destination);

// ── Play incoming audio ──────────────────────────────────────────
ws.onmessage = async (event) => {
  if (event.data instanceof ArrayBuffer) {
    // Decode PCM 24 kHz and play
    const audioBuffer = await ctx.decodeAudioData(event.data);
    const playback    = ctx.createBufferSource();
    playback.buffer   = audioBuffer;
    playback.connect(ctx.destination);
    playback.start();
  } else {
    const data = JSON.parse(event.data);
    if (data.type === "transcript") {
      document.getElementById("transcript").textContent = data.text;
    }
  }
};
